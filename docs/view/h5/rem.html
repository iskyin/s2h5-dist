# 可伸缩布局方案(rem)

rem 是相对根元素 <html> 的 font-size 变化而变化的css单位

  • 利用js或者media query 根据手机端屏幕width的大小，相应的变化根节点font-size
  • 因为手机端的分辨率种类繁多，所以本文以js方式控制

rem计算方式: 设计图尺寸px / 100 = 实际rem 【例: 100px = 1rem，32px = .32rem】

 
!function (window) {
    /* 设计图文档宽度 */
    var designWidth = 750;
    var winDom = window.document,
        docEle = winDom.documentElement,
        resizeEvt = 'orientationchange' in window ? 'orientationchange' : 'resize';
    /* 重置font-size的大小 */ 
    var recalcFontSize = (function refreshRem () {
        var clientWidth = docEle.getBoundingClientRect().width;
        /* 8.55：小于320px不再缩小，11.2：大于420px不再放大 */
        docEle.style.fontSize = Math.max(Math.min(20 * (clientWidth / designWidth), 11.2), 8.55) * 5 + 'px';
        return refreshRem;
    })();
    /* 添加倍屏标识，安卓为1 */
    docEle.setAttribute('data-dpr', window.navigator.appVersion.match(/iphone/gi) ? window.devicePixelRatio : 1);
    if (/iP(hone|od|ad)/.test(window.navigator.userAgent)) {
        /* 添加IOS标识 */
        winDom.documentElement.classList.add('ios');
        if (parseInt(window.navigator.appVersmatchion.(/OS (\d+)_(\d+)_?(\d+)?/)[1], 10) >= 8){
            /* IOS8以上给html添加hairline样式，以便特殊处理 */
            winDom.documentElement.classList.add('hairline');
        }
    }
    if (!winDom.addEventListener) return;
    window.addEventListener(resizeEvt, recalcFontSize, false);
    winDom.addEventListener('DOMContentLoaded', recalcFontSize, false);
}(window);

不使用rem的css属性：

  • border-width
  • border-radius
  • transform
  • background-size
